  3 /**
  4  * @class
  5  *
  6  * tuple represents the collection tuple as managed by the the
  7  * tao matching api
  8  *
  9  * @access public

 12  * @extends TAO_MATCHING.Collection
 13  * 
 14  * @constructor 
 15  * @param array data data used to construct the tuple
 16  */
 17 TAO_MATCHING.Tuple = function (data) {
 18 	// Call the parent constructor
 19 	TAO_MATCHING.Collection.call (this);
 20     // The length of the tuple
 21     this.count = 0;
 22     // The tuple's values
 23     this.value = {};
 24 	// Set the value of the variable
 25 	this.setValue (data);
 26 };

156     , toJSon : function ()
157     {        
158         var returnValue = {};
159         
160         for (var i in this.value) {
161             var tmpValue = this.value[i];
162             
163             if (TAO_MATCHING.VariableFactory.isMatchingVariable(tmpValue)){
164                 tmpValue = tmpValue.toJSon();
165             }
166             returnValue[i] = tmpValue;
167         }
168         
169         return returnValue;
170     }
