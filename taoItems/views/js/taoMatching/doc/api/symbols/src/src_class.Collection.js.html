 15  * @constructor 
 16  */
 17 TAO_MATCHING.Collection = function (data) {
 18 	// Call the parent constructor
 19 	TAO_MATCHING.Variable.call (this);
 20 };
 21 
 22 TAO_MATCHING.Collection.prototype = {
 23     /**
 24      * Check if the collection contains the given element.
 25      *
 26      * @access public
 27      * @author Cedric Alfonsi, <cedric.alfonsi@tudor.lu>
 28      * @param  Variable var
 29      * @param  array options options.needleType {String} Define if the script has to go through the needle or it must treat it as  scalar variable
 30      * @return boolean
 31      */
 32     contains : function (matchingVar, options)
 33     {
 34 		var returnValue = null;
 35 		var needleType = 'collection';
 36 
 37 		if (typeof options != 'undefined'){
 38 		    if (typeof options.needleType != 'undefined'){
 39                 needleType = options.needleType;
 40             }
 41 		}
 42 		
 43 		// If the needle is a Tuple
 44         if (TAO_MATCHING.Variable.isTuple (matchingVar) && needleType == 'collection') {
 45             returnValue = false;
 46             for (var key in matchingVar.value) {
 47                 if (this.value[key].match(matchingVar.value[key])){
 48                     returnValue = true;
 49                 }else{
 50                     returnValue = false;
 51                     break;
 52                 }
 53             }
 54         }
 55         // Else if the needle is a List
 56         else if (TAO_MATCHING.Variable.isList (matchingVar) && needleType == 'collection') {
 57             returnValue = false;
 58             for (var key in matchingVar.value) {
 59                 if (this.contains (matchingVar.value[key]), {"matchingVarList":false}){
 60                     returnValue = true;
 61                 } else {
 62                     returnValue = false;
 63                     break;
 64                 }
 65             }
 66         } 
 67         // Else we check if the value is include is the current collection
 68         else {
 69             returnValue = false;
 70             
 71             for (var key in this.value) {
 72                 // If the needle is not of the same type that an item of the collection (escape)
 73                 if (matchingVar.getType() != this.value[key].getType()){
 74                     returnValue = false;
 75                     break;
 76                 } 
 77                 // Else we check if the needle match the current item
 78                 else if (matchingVar.match (this.value[key])) {
 79                     returnValue = true;
 80                     break;
 81                 } else {
 82                     returnValue = false;
 83                 }
 84             }
 85         }
 86         
 87         return returnValue;
 88     }
 89 	
 90 	/**
 91 	 * check if the variable is null
 92 	 *
 93 	 * @access public
 94 	 * @author Cedric Alfonsi, <cedric.alfonsi@tudor.lu>
 95 	 * @return boolean
 96 	 */
 97     , isNull : function ()
 98     {
 99 		if (this.value == null)
100 			return true;
101 		if (this.length() == 0)
102         	return true;
103 		return false;
104     }

106 };
