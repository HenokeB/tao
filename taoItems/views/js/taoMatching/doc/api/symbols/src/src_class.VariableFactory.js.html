 13 TAO_MATCHING.VariableFactory = {

 15     /**
 16      * Create a variable functions of the given data.
 17      *
 18      * @access public
 19      * @author Cedric Alfonsi, <cedric.alfonsi@tudor.lu>
 20      * @param  array data Data of the variable (the value)
 21      * @param  string type The type is optional, if it is not defined the data will
 22 define the type of the variable
 23      * @return TAO_MATCHING.Variable
 24      */
 25 	create : function (data, pType) {
 26 		var returnValue = null;
 27 		var type = null;
 28 		
 29 		// The type has been defined manually
 30 		if (typeof pType != 'undefined' && pType != null) {
 31             type = pType;
 32 		}
 33 		else {
 34             type = TAO_MATCHING.VariableFactory.getType (data);
 35 		}
 36 
 37 		// Create the variable according to its type
 38 		switch (type) {
 39 			// Collection Tuple : our standard defines an JSON Object as a tuple
 40 			case 'tuple':
 41 			case 'point':
 42 				returnValue = new TAO_MATCHING.Tuple (data);
 43 				break;
 44 				
 45 			// Collection List : our standard defines a JSON Array as a list
 46 			case 'list':
 47 				returnValue = new TAO_MATCHING.List (data);
 48 				break;
 49 			
 50 			// Native language variable types			
 51 			case 'boolean':
 52 			case 'number':
 53 			case 'string':
 54 			case 'NULL':
 55 				returnValue = new TAO_MATCHING.BaseTypeVariable (data);
 56 				break;
 57 
 58             // Shape       
 59             case 'circle':
 60             case 'ellipse':
 61                 returnValue = new TAO_MATCHING.Ellipse (data);
 62                 break;
 63             case 'rect':
 64             case 'poly':
 65                 returnValue = new TAO_MATCHING.Poly (data);
 66                 break;
 67 		
 68 		    // The type is not supported by the matching API
 69 			default:
 70 				throw new Error ('TAO_MATCHING.VariableFactory::create variable type unknown '+type+' for '+data);
 71 		}
 72 		
 73         return returnValue;
 74 	}
 75 	
 76 	/** 
 77 	 * Define the type of data from the data value
 78 	 * @params {object} data Get the type of these data
 79 	 * @return string
 80 	 * @private
 81 	 */
 82 	, getType : function (data){
 83 	    var returnValue = null;
 84 	    
 85 	    // We get an object as data
 86         if (typeof data == 'object') {
 87             // If the data is null, we create a basic type variable with null value
 88             if (data == null) {
 89                 returnValue = 'NULL';
 90             }
 91             // If a type as been declared
 92             else if (data.type!=undefined){
 93                 returnValue  = data.type;
 94             }
 95             // If the data is an array we create a list variable
 96             else if (($.isArray(data))) {
 97                 returnValue = 'list';
 98             } 
 99             // If the data is an object, our format define this data as a tuple
100             else {
101                 returnValue = 'tuple';
102             }
103         }
104         else {
105             returnValue = typeof data; 
106         }
107         
108         return returnValue;
109 	}

111     /**
112      * Check if the variable is a Matching Variable (deal with javascript)
113      *
114      * @access public
115      * @author Cedric Alfonsi, <cedric.alfonsi@tudor.lu>
116      * @param object data Check if the given data is a variable of the matching api 
117      * @return boolean
118      */	
119 	, isMatchingVariable : function (data){
120 	    var returnValue = false;
121 	    
122 	    if (data instanceof TAO_MATCHING.BaseTypeVariable
123             || data instanceof TAO_MATCHING.Tuple
124             || data instanceof TAO_MATCHING.List)
125         {
126             returnValue = true;
127         }
128 	    
129 	    return returnValue;
130 	}

132     /**
133      * Convert data in numeric BaseTypeVariable.
134      * If the data is not a valid base type value return null.
135      *
136      * @access public
137      * @author Cedric Alfonsi, <cedric.alfonsi@tudor.lu>
138      * @param  data Data to convert
139      * @return TAO_MATCHING.BaseTypeVariable
140      */
141      , toNumericBaseType : function (data) {
142         var returnValue = null;
143         
144         // IF the first expression is not a BaseTypeVariable try to create it
145         if (!(data instanceof TAO_MATCHING.BaseTypeVariable)){
146             if (TAO_MATCHING.BaseTypeVariable.isValidValue (data)) {
147                 var matchingVar = new TAO_MATCHING.BaseTypeVariable (data);
148                 if (matchingVar.isNumerical ()){
149                     returnValue = matchingVar;
150                 }
151             }
152         } else {
153             if (data.isNumerical()){
154                 returnValue = data;
155             }
156         }
157         
158         return returnValue;
159      }

165       * @access public

167       * @param  data Data to convert
168       * @return  TAO_MATCHING.BaseTypeVariable
169       */
170      , toBooleanBaseType : function (data) {

174         if (!(data instanceof TAO_MATCHING.BaseTypeVariable)){
175             if (TAO_MATCHING.BaseTypeVariable.isValidValue (data)) {

177                 if (matchingVar.isBoolean ()){
178                     returnValue = matchingVar;
179                 }
180             }
181         } else {
