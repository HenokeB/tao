  3 /**
  4  * @class
  5  * 
  6  * Poly represents a polygon managed by the system.
  7  * A polygon hosts a list points. 
  8  * A point is represented on the system by a "Tuple Variable"
  9  * of two integers.
 10  *
 11  * @extends TAO_MATCHING.Shape
 12  * @author Cedric Alfonsi, <cedric.alfonsi@tudor.lu>
 13  * @package TAO_MATCHING
 14  * 
 15  * @constructor 
 16  * @param {array} object Array of points
 17  */
 18 TAO_MATCHING.Poly = function(data) {
 19     // Call the parent constructor
 20     TAO_MATCHING.Shape.call (this, data.type);
 21     
 22     /**
 23      * List of points which form the polygon
 24      *
 25      * @access protected
 26      * @var array
 27      */
 28     this.points = [];
 29     
 30     // treat the data
 31     for (var i in data.points) {
 32         this.points.push (TAO_MATCHING.VariableFactory.create(data.points[i]));
 33     }
 34 };
 35 
 36 TAO_MATCHING.Poly.prototype = {
 37     /**
 38      * Check if the polygon contains a point
 39      *
 40      * @access public
 41      * @author Cedric Alfonsi, <cedric.alfonsi@tudor.lu>
 42      * @param  Variable var The point to find.
 43 	 * The point is represented by a "Tuple Variable" of two integers.
 44      * @return boolean
 45      */
 46     contains : function (point) {
 47         var returnValue = null;
 48 
 49         var point_x = (point.getValue())[0].getValue();
 50         var point_y = (point.getValue())[1].getValue();
 51 
 52         // Algorithm from http://jsfromhell.com/math/is-point-in-poly
 53         for(var c = false, i = -1, l = this.points.length, j = l - 1; ++i < l; j = i) {
 54             var point_i_x = (this.points[i].getValue())[0].getValue();
 55             var point_i_y = (this.points[i].getValue())[1].getValue();
 56             var point_j_x = (this.points[j].getValue())[0].getValue();
 57             var point_j_y = (this.points[j].getValue())[1].getValue();
 58 
 59             ( ((point_i_y <= point_y) && (point_y < point_j_y))
 60                 || ((point_j_y <= point_y ) && (point_y < point_i_y))
 61             )
 62             && (point_x < ( (point_j_x - point_i_x) * (point_y - point_i_y) / (point_j_y - point_i_y) + point_i_x))
 63             && (c = !c);
 64         }
 65         
 66         returnValue = c;
 67         return returnValue;
 68     }
 69 };
