1 TAO_MATCHING = typeof TAO_MATCHING != 'undefined' ? TAO_MATCHING : {};
  2 /**
  3  * @class
  4  * 
  5  * Ellipse represents ellipse managed by the tao matching api.
  6  * Ellipse is used to represent ellipse and circle.
  7  *
  8  * @abstract
  9  * @author Cedric Alfonsi, <cedric.alfonsi@tudor.lu>
 10  * @package TAO_MATCHING
 11  * @todo Check if the function setTypes is deprecated
 12  * 
 13  * @constructor 
 14  * @param {array} data Data used to construct the ellipse
 15  */
 16 TAO_MATCHING.Ellipse = function(data){
 17     // Call the parent constructor
 18     TAO_MATCHING.Shape.call (this, data.type);
 19     /**
 20      * Center of the Ellipse
 21      *
 22      * @access protected
 23      * @type {TAO_MATCHING.Tuple}
 24      */
 25     this.center = TAO_MATCHING.VariableFactory.create(data.center);
 26     /**
 27      * Horizontal radius of the ellipse
 28      *
 29      * @access protected
 30      * @var Integer
 31      */
 32     this.hradius = data.hradius;
 33     /**
 34      * Vertical radius of the ellipse
 35      *
 36      * @access protected
 37      * @var Integer
 38      */
 39     this.vradius = data.vradius;
 40 };
 41 
 42 TAO_MATCHING.Ellipse.prototype = {
 43     /**
 44      * Check if the polygon contains a point
 45      * @access public
 46      * @author Cedric Alfonsi, <cedric.alfonsi@tudor.lu>
 47      * @param {TAO_MATCHING.Variable} point The point to find. The point is represented by a "Tuple Variable" of two integers.
 48      * @return boolean
 49      */
 50     contains : function (point) {
 51         var returnValue = false;
 52         var point_x = (point.getValue())[0].getValue();
 53         var point_y = (point.getValue())[1].getValue();
 54         var center_x = (this.center.getValue())[0].getValue();
 55         var center_y = (this.center.getValue())[1].getValue();
 56         var a = Math.pow(point_x-center_x,2)/Math.pow(this.hradius,2);
 57         var b = Math.pow(point_y-center_y,2)/Math.pow(this.vradius,2);
 58         returnValue = a+b <= 1;
 59         return returnValue;
 60     }
 61 };

 63 // Extend the class with its parent properties
 64 TAO_MATCHING.Ellipse.prototype = $.extend ({}, TAO_MATCHING.Shape.prototype, TAO_MATCHING.Ellipse.prototype);
