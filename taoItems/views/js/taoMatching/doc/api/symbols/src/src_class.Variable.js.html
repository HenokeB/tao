126 /**
127  * Check if a value is a scalar value
128  * @param {object} data Data to check
129  * @return boolean
130  * @static
131  */
132 TAO_MATCHING.Variable.isScalar = function (data) {
133     var returnValue = false;
134     switch (typeof data){
135         case "boolean":
136         case "number":
137         case "string":
138             returnValue = true;
139     }
140     return returnValue;
141 };

143 /**
144  * Check if a value is a collection value
145  * @param {object} data Data to check
146  * @return boolean
147  * @static
148  */
149 TAO_MATCHING.Variable.isCollection = function (data) {
150     var returnValue = false;
151     
152     if (typeof data.getType != 'undefined') {
153         switch (data.getType()){
154             case "list":
155             case "tuple":
156                 returnValue = true;
157         }
158     }
159     
160     return returnValue;
161 };
