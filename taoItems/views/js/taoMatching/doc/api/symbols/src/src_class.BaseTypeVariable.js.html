  3 /**
  4  * @class
  5  * 
  6  * BaseTypeVariable represents the scalar variables
  7  * managed by the system.
  8  *
  9  * @access public
 10  * @author Cedric Alfonsi, <cedric.alfonsi@tudor.lu>
 11  * @package TAO_MATCHING
 12  * @extends TAO_MATCHING.Variable
 13  * 
 14  * @constructor 
 15  * @param object value Value of the variable
 16  */
 17 TAO_MATCHING.BaseTypeVariable = function(data){
 18 	// Call the parent constructor
 19 	TAO_MATCHING.Variable.call (this);
 20 	// Set the value of the variable
 21 	this.setValue (data);
 22 }

 24 TAO_MATCHING.BaseTypeVariable.prototype = {
 25 	    /**
 26 	     * Get the type of the variable
 27 	     *
 28 	     * @access public
 29 	     * @author Cedric Alfonsi, <cedric.alfonsi@tudor.lu>
 30 	     * @return string
 31 	     */
 32     getType : function ()
 33     {
 34         return (typeof this.value);
 35     }
 36 
 37     /**
 38      * Compare the variable with an other
 39      *
 40      * @access public
 41      * @author Cedric Alfonsi, <cedric.alfonsi@tudor.lu>
 42      * @param  Variable var Variable to compare
 43      * @return boolean
 44      */
 45     , equal : function (matchingVar)
 46     {
 47 		if (! (matchingVar instanceof TAO_MATCHING.BaseTypeVariable))
 48 			throw new Error ('TAO_MATCHING.BaseTypeVariable.equal an error occured : first argument expected type TAO_MATCHING.BaseTypeVariable, given : '+(typeof matchingVar));
 49 		
 50         var returnValue = false;
 51 
 52         if (this.getType() != matchingVar.getType()){
 53         	returnValue = false;
 54         } else {
 55         	returnValue = (this.getValue() == matchingVar.getValue());	
 56         }
 57         		
 58         return returnValue;
 59     }

 61     /**

 64      * @access public

 67      * @see Variable

162 /**
163  * isValidValue function provide a way to test if a value could be BaseTypeVariable value type compliant
164  *
165  * @access public
166  * @static
167  * @author Cedric Alfonsi, <cedric.alfonsi@tudor.lu>
168  * @param  data
169  * @return boolean
170  */
171 TAO_MATCHING.BaseTypeVariable.isValidValue = function (data)
172 {
173     var returnValue = false;
174     
175     // If the data is null
176     if (data == null){
177         returnValue = true;
178     }
179     else {
180         // If the data has a valid type
181         switch (typeof (data)){
182             case 'boolean':
183             case 'number':
184             case 'string':
185                 returnValue = true;
186         }    
187     }
188 
189     return returnValue;
190 };
