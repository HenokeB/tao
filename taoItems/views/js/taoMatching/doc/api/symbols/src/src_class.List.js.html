  3 /**
  4  * @class

  7  * tao matching api

 12  * @extends TAO_MATCHING.Collection
 13  * 
 14  * @constructor 
 15  * @param  array data data used to construct the tuple
 16  */
 17 TAO_MATCHING.List = function (data) {
 18 	// Call the parent constructor
 19 	TAO_MATCHING.Collection.call (this);
 20 	// Set the value of the variable
 21 	this.setValue (data);
 22 };

 67     /**
 68      * Match a list with another
 69      *
 70      * @access public
 71      * @author Cedric Alfonsi, <cedric.alfonsi@tudor.lu>
 72      * @param  List list
 73      * @return boolean
 74      */
 75     , match : function (list)
 76     {
 77 		
 78 		if (! (list instanceof TAO_MATCHING.List))
 79 			throw new Error ('TAO_MATCHING.List.match an error occured : first argument expected type TAO_MATCHING.List, given : '+(typeof list));
 80 
 81         var returnValue = true;
 82         		
 83         // If the cardinality is not the same return false
 84         if (this.length() != list.length()){
 85         	return false;	
 86         }
 87         
 88         // Test if the both lists have the same content
 89         for (var i=0; i<this.length(); i++) {
 90 	        var tempResult = false;
 91 			
 92 	        for (var j=0; j<list.length(); j++) {
 93 	        	if (this.getElement(i).getType () != list.getElement(j).getType()){
 94 	        		throw new Exception ('TAO_MATCHING.List::match an error occured : types of the elements to match are not the same ['+ this.getElement(i).getType () +'] and ['+ list.getElement(j).getType() +']');
 95 	        		returnValue = false;
 96 	        	} 
 97 	        	else if (this.getElement(i).match(list.getElement(j))) {
 98 	            	tempResult = true;
 99 	                break;
100 	            } 
101 				else {
102 					// DOES NOT MATCH
103 				}
104 	        }
105 	        if (!tempResult){
106 	        	returnValue = false;
107 				break;	
108 			}
109         }
110 
111         return returnValue;
112     }

143     /**
144      * Export the variable in jSon format.
145      * {
146      *     "identifier":"myVariableIdentifier",
147      *     "value": [
148      *         "myVar1"
149      *         "myVar2"
150      *     ]
151      * }
152      *
153      * @access public
154      * @author Cedric Alfonsi, <cedric.alfonsi@tudor.lu>
155      */
156     , toJSon : function ()
157     {
158         var returnValue = Array ();
159         
160         for (var i in this.value) {
161             var tmpValue = this.value[i];
162             if (TAO_MATCHING.VariableFactory.isMatchingVariable(tmpValue)){
163                 tmpValue = tmpValue.toJSon();
164             }
165             returnValue [i] = tmpValue
166         }
167         
168         return returnValue;
169     }
